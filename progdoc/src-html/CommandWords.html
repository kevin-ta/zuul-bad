001import java.util.HashMap;
002import java.util.Set;
003
004/**
005 * This class is part of the "World of Zuul" application. 
006 * "World of Zuul" is a very simple, text based adventure game.  
007 * 
008 * This class holds an enumeration table of all command words known to the game.
009 * It is used to recognise commands as they are typed in.
010 *
011 * @author  Michael Kolling and David J. Barnes + D.Bureau
012 * @version 2008.03.30 + 2013.09.15
013 */
014public class CommandWords
015{
016    private HashMap<String, CommandWord> validCommands;
017
018    /**
019     * Constructeur par defaut
020     */
021    public CommandWords()
022    {
023        validCommands = new HashMap<String, CommandWord>();
024        for(CommandWord command : CommandWord.values()) {
025            if(command != CommandWord.UNKNOWN) {
026                validCommands.put(command.toString(), command);
027            }
028        }
029    } // CommandWords()
030
031    /**
032     * Verifie si une String donnee fait partie des commandes valides. 
033     * @param pString la String a tester
034     * @return true si pString est une comande valide, false sinon
035     */
036    public boolean isCommand(final String pString)
037    {
038        return validCommands.containsKey(pString);
039    } // isCommand()
040    
041    /**
042     * Affiche tous les commandes valides
043     */
044    public String getCommandList() {
045        String vCommandList = new String();
046        Set<String> keys = validCommands.keySet();
047        for(String commands : keys) {
048            vCommandList += " " + commands;
049        }
050        return vCommandList;
051    }
052    
053    /**
054     * Find the CommandWord associated with a command word.
055     * @param commandWord The word to look up.
056     * @return The CommandWord correspondng to commandWord, or UNKNOWN
057     *         if it is not a valid command word.
058     */
059    public CommandWord getCommandWord(String commandWord)
060    {
061        CommandWord command = validCommands.get(commandWord);
062        if(command != null) {
063            return command;
064        }
065        else {
066            return CommandWord.UNKNOWN;
067        }
068    }
069} // CommandWords
