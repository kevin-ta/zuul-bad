001package pkg_commands;
002
003import java.util.StringTokenizer;
004
005/**
006 * This class is part of the "World of Zuul" application. 
007 * "World of Zuul" is a very simple, text based adventure game.  
008 * 
009 * This parser reads user input and tries to interpret it as an "Adventure"
010 * command. Every time it is called it reads a line from the terminal and
011 * tries to interpret the line as a two word command. It returns the command
012 * as an object of class Command.
013 *
014 * The parser has a set of known command words. It checks user input against
015 * the known commands, and if the input is not one of the known commands, it
016 * returns a command object that is marked as an unknown command.
017 * 
018 * @author  Michael Kolling and David J. Barnes + D.Bureau
019 * @version 2008.03.30 + 2013.09.15
020 */
021public class Parser 
022{
023    private CommandWords aValidCommands;
024
025    /**
026     * Constructeur de la classe Parser qui cree les 2 objets prevus pour les attributs
027     */
028    public Parser() 
029    {
030        this.aValidCommands = new CommandWords();
031    }
032
033    /**
034     * Get a new command from the user. The command is read by parsing the 'inputLine'.
035     * @param inputLine A string containing the inputline
036     * @return A new instantiated Command object
037     */
038    public Command getCommand(String inputLine) 
039    {
040        //String inputLine = "";   // will hold the full input line
041        String word1;
042        String word2;
043
044        StringTokenizer tokenizer = new StringTokenizer(inputLine);
045
046        if(tokenizer.hasMoreTokens())
047            word1 = tokenizer.nextToken();      // get first word
048        else
049            word1 = null;
050        if(tokenizer.hasMoreTokens())
051            word2 = tokenizer.nextToken();      // get second word
052        else
053            word2 = null;
054
055        // note: we just ignore the rest of the input line.
056
057        // Now check whether this word is known. If so, create a command
058        // with it. If not, create a "null" command (for unknown command).
059        
060        Command command = this.aValidCommands.get(word1);
061        if(command != null) {
062            command.setSecondWord(word2);
063        }
064        return command;
065    }
066    
067    /**
068     * Fonction retournant une chaine de caracteres contenant la liste des commandes disponibles
069     * @return Une chaine de caracteres contenant la liste des commandes disponibles
070     */
071    public String getCommandList()
072    {
073        return aValidCommands.getCommandList();
074    }
075}
