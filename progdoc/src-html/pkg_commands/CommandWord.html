001package pkg_commands;
002
003import pkg_game.GameEngine;
004
005/**
006 * Representations for all the valid command words for the game.
007 * 
008 * @author Michael Kolling and David J. Barnes
009 * @version 2006.03.30
010 */
011public enum CommandWord
012{
013    GO("go", new GoCommand()),
014    QUIT("quit", new QuitCommand()),
015    HELP("aider", new HelpCommand()),
016    LOOK("look", new LookCommand()),
017    EAT("eat", new EatCommand()),
018    BACK("back", new BackCommand()),
019    TEST("test", new TestCommand()),
020    TAKE("take", new TakeCommand()),
021    DROP("drop", new DropCommand()),
022    ITEMS("items", new ItemsCommand()),
023    CHARGE("charge", new ChargeCommand()),
024    TELEPORT("teleport", new TeleportCommand()),
025        ALEA("alea", new AleaCommand()),
026        TALK("talk", new TalkCommand()),
027    UNKNOWN("?", null);
028    private String commandString;
029    private Command command;
030    
031    /**
032     * Initialise with the corresponding command word.
033     * @param commandString The command string.
034     * @param pCommand The Command class
035     */
036    CommandWord(String commandString, Command pCommand)
037    {
038        this.commandString = commandString;
039        this.command = pCommand;
040    }
041    
042    /**
043     * Fonction permettant de renvoyer la commande en tant que chaine de caracteres
044     * @return The command word as a string.
045     */
046    public String toString()
047    {
048        return commandString;
049    }
050    
051    /**
052     * Gets the command
053     *
054     * @return The command
055     */
056    public Command getCommand()
057    {
058        return this.command;
059    }
060}
