001package pkg_commands;
002
003import pkg_game.GameEngine;
004import pkg_game.Player;
005import pkg_item.Item;
006
007/**
008 * Classe de la commande take
009 * 
010 * @author Kevin Ta
011 * @version 2015 - 2016
012 */
013public class TakeCommand extends Command
014{
015    /**
016     * Constructeur de la classe TakeCommand
017     */
018    public TakeCommand()
019    {
020    }
021    
022    /**
023     * Fonction permettant d'executer la commande take
024     * @param player Un objet Player
025     * @return false
026     */
027    public boolean execute(Player player)
028    {
029        if(!this.hasSecondWord()) 
030        {
031            GameEngine.gui.println("Take what?\n");
032            return false;
033        }
034        
035        String name = this.getSecondWord();
036        Item item = (player.getCurrentRoom()).findItem(name);
037
038        if (item != null)
039        {
040            double weight = item.getWeight();
041            if (player.getMaxWeight() - weight >= 0)
042            {
043                player.getCurrentRoom().removeItem(item);
044                player.addItem(item);
045                player.setMaxWeight(-weight);
046            }
047            else
048            {
049                GameEngine.gui.println("You can't take more.\n");
050            }
051        }     
052        else
053        {
054            GameEngine.gui.println("The item doesn't exist.\n");
055        }
056        return false;
057    }
058}
