001import java.util.Stack;
002
003/**
004 * Classe principale du jeu, elle permet de démarrer Zuul.
005 */
006public class GameEngine
007{
008    private Room aCurrentRoom;
009    private Parser aParser;
010    private UserInterface gui;
011    public static Stack<Room> roomHistory;
012
013    /**
014     * Constructeur de la classe Game.
015     * Il permet d'initialiser le jeu et d'afficher un message de bienvenue
016     */
017    public GameEngine()
018    {
019        createRooms();
020        this.aParser = new Parser();
021        roomHistory = new Stack<Room>();
022    }
023    
024    /**
025     * Création du gui
026     */
027    public void setGUI(UserInterface userInterface)
028    {
029        gui = userInterface;
030        printWelcome();
031    }
032    
033    /**
034     * Procédure permettant de créer les pièces du jeu
035     */
036    private void createRooms()
037    {
038        Room vOutside = new Room("outside the main entrance of the university", "images/outside.gif");
039        Room vTheatre = new Room("in a lecture theatre", "images/courtyard.gif");
040        Room vPub     = new Room("in the campus pub", "images/dungeon.gif");
041        Room vLab     = new Room("in a computing lab", "images/hallway.gif");
042        Room vServer  = new Room("in the server room", "images/stairs.gif");
043        Room vOffice  = new Room("in the computing admin office", "images/castle.gif");
044        
045        vOutside.setExit("east", vTheatre);
046        vOutside.setExit("west", vPub);
047        vOutside.setExit("south", vLab);
048        
049        vTheatre.setExit("west", vOutside);
050        
051        vPub.setExit("east", vOutside);
052        
053        vLab.setExit("north", vOutside);
054        vLab.setExit("east", vOffice);
055        vLab.setExit("up", vServer);
056        
057        vServer.setExit("down", vLab);
058
059        vOffice.setExit("west", vLab);
060        
061        vOutside.addItem(new Item("beer", 0.5));
062        vOutside.addItem(new Item("cola", 1));
063        vTheatre.addItem(new Item("juice", 1));
064
065        this.aCurrentRoom = vOutside;
066    }
067    
068    /**
069     * Given a command, process (that is: execute) the command.
070     * If this command ends the game, true is returned, otherwise false is
071     * returned.
072     */
073    public void interpretCommand(String commandLine) 
074    {
075        gui.println(commandLine);
076        Command command = aParser.getCommand(commandLine);
077
078        if(command.isUnknown()) {
079            gui.println("I don't know what you mean...");
080            return;
081        }
082
083        String commandWord = command.getCommandWord();
084        if (commandWord.equals("help"))
085            printHelp();
086        else if (commandWord.equals("go"))
087            goRoom(command);
088        else if (commandWord.equals("quit")) {
089            if(command.hasSecondWord())
090                gui.println("Quit what?");
091            else
092                endGame();
093        }
094        else if (commandWord.equals("eat"))
095            eat();
096        else if (commandWord.equals("look"))
097            look();
098        else if (commandWord.equals("back"))
099            goBack(command);
100    }
101    
102    /**
103     * Affiche des informations sur la pièce actuelle
104     */
105    private void printLocationInfo()
106    {
107        gui.println(this.aCurrentRoom.getLongDescription() + "\n");
108    }
109    
110    /**
111     * Affiche le message de bienvenue
112     */
113    private void printWelcome()
114    {
115        gui.println("Welcome to the World of Zuul!");
116        gui.println("World of Zuul is a new, incredibly boring adventure game.");
117        gui.println("Type 'help' if you need help.\n");
118        printLocationInfo();
119        gui.showImage(this.aCurrentRoom.getImageName());
120    }
121    
122    /**
123     * Commande de déplacement
124     */
125    private void goRoom(final Command pCommand)
126    {
127        if (!pCommand.hasSecondWord())
128        {
129            gui.println("Go where ?");
130            return;
131        }
132        String vDirection = pCommand.getSecondWord();
133        Room vNextRoom = this.aCurrentRoom.getExit(vDirection);
134        if (vNextRoom == null)
135        {
136            gui.println("There is no door !");
137        }
138        else
139        {
140            roomHistory.push(this.aCurrentRoom);
141            this.aCurrentRoom = vNextRoom;
142            printLocationInfo();
143            if(aCurrentRoom.getImageName() != null)
144                gui.showImage(aCurrentRoom.getImageName());
145        }
146    }
147    
148    /**
149     * Commande d'aide
150     */
151    private void printHelp()
152    {
153        gui.println("You are lost. You are alone.");
154        gui.println("You wander around at the university.");
155        gui.println("Your command words are:");
156        gui.println(this.aParser.getCommandList() + "\n");
157    }
158    
159    /**
160     * Affiche des informations sur la pièce actuelle
161     */
162    private void look()
163    {
164        gui.println(this.aCurrentRoom.getLongDescription() + "\n");
165    }
166    
167    /**
168     * Commande eat
169     */
170    private void eat()
171    {
172        gui.println("You have eaten now and you are not hungry any more.\n");
173    }
174    
175    /**
176     * Permet de revenir en arrière
177     */
178    private void goBack(final Command pCommand)
179    {
180        if (pCommand.hasSecondWord())
181        {
182            gui.println("Back where ?");
183            return;
184        }
185        if (roomHistory.isEmpty())
186        {
187            gui.println("You can't back anymore");
188        }
189        else
190        {
191            this.aCurrentRoom = roomHistory.pop();
192            printLocationInfo();
193            if(aCurrentRoom.getImageName() != null)
194                gui.showImage(aCurrentRoom.getImageName());
195        }
196    }
197    
198    /**
199     * Met fin au jeu
200     */
201    private void endGame()
202    {
203        gui.println("Thank you for playing.  Good bye.");
204        gui.enable(false);
205    }
206} // Game
